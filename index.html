	Abaixo existe dois códigos GML para o menu do game test que foi criado recentemente (e publicado aqui).
	Existe um problema de interação entre 'mouse' e 'teclado'.

	Só é preciso deixar as funções já existentes de mouse e teclado 'somadas e funcionais'.
		

	/// @description Menu Criar

	window_set_size(1280, 720)

	options = ["novo jogo","carregar jogo","opções","sair"];
	index = 0;
	op_max = array_length(options);
		

	/// @description Menu Draw

	// Variáveis para as escalas do mouse e do teclado
var _mx = device_mouse_x_to_gui(0); // Obtém a posição X atual do mouse em relação ao canto superior esquerdo da janela de jogo convertida em coordenadas de GUI.
var _my = device_mouse_y_to_gui(0); // Obtém a posição Y atual do mouse em relação ao canto superior esquerdo da janela de jogo convertida em coordenadas de GUI.

// Definição de fonte e alinhamento
draw_set_font(ftMenu); // Define a fonte atual para desenho de texto
draw_set_halign(fa_center); // Define o alinhamento horizontal do texto para centralizado
draw_set_valign(fa_middle); // Define o alinhamento vertical do texto para centralizado

var tam_menu = array_length(menu_inicial); // Obtém o comprimento do array "menu_inicial" e armazena em "tam_menu".

function sel_hover(mx, my, x1, y1, x2, y2, selectedColor, hoverColor) {
    if (point_in_rectangle(mx, my, x1, y1, x2, y2)) {
        draw_set_color(hoverColor);
    } else {
        draw_set_color(selectedColor);
    }
}

// Loop que desenha o menu
for(var i = 0; i < tam_menu; i++)
{
	// Definição de espaço entre linhas
	var gui_largura = display_get_gui_width(); // Obtém a largura atual da janela de jogo convertida em coordenadas de GUI
	var gui_altura = display_get_gui_height(); // Obtém a largura atual da janela de jogo convertida em coordenadas de GUI
	var altura_str = string_height("I"); // Obtém a altura aproximada da string "I" com a fonte atual
	var largura_str = string_width(menu_inicial[i]); // Obtém a largura aproximada da string atual do menu com a fonte atual

	// Define as coordenadas de canto superior esquerdo e canto inferior direito do retângulo atual do menu
	var x1 = gui_largura / 2 - largura_str / 2;
	var y1 = gui_altura / 2 - altura_str / 2 + altura_str * i; //top left MOUSE

	var x2 = gui_largura / 2 + largura_str / 2;
	var y2 = gui_altura / 2 + altura_str / 2 + altura_str * i; // top right MOUSE

	sel_hover(_mx, _my, x1, y1, x2, y2, c_white, c_orange); // adiciona a função sel_hover para mudar a cor do texto

	if(point_in_rectangle(_mx, _my, x1, y1, x2, y2)) // Verifica se a posição atual do mouse está dentro do retângulo do menu atual
	{
		escala[i] = lerp(escala[i], 1.4, 0.15);	// Interpola linearmente a escala atual para 1.4 (aumento da escala) com uma velocidade de interpolação de 0.15 por quadro.

		if(mouse_check_button_pressed(mb_left)) // Verifica se o botão esquerdo do mouse foi pressionado
		{
			switch menu_inicial[i] // Verifica qual opção do menu foi selecionada
			{
				case menu_inicial[0]: //Jogar
					/*show_message("Vamos Jogar");*/
					room_goto(rLevel1) // Vai para a room (sala) com o nome "rLevel1"
					break; // sai do switch
				case menu_inicial[1]: //Carregar
					show_message("Carregar um jogo salvo"); // Exibe uma mensagem na tela
					break; // sai do switch
				case menu_inicial[2]: //Opções
					show_message("Tenho que configurar algumas coisas");
					break; // sai do switch
				case menu_inicial[3]: //Sair - verifica se o último item do menu foi selecionado
					game_end(); // encerra o jogo
					break; // sai do switch
			}
		}
	}
	else // se o mouse não estiver sobre o item do menu
	{
		escala[i] = lerp(escala[i], 1, 0.15); // ajusta a escala para 1 (sem aumento)
	}

	if(index == i) // se o item atual é o selecionado
	{
		draw_set_color(c_orange); // altera a cor do texto para laranja
	}
	else // se não é o item selecionado
	{
		draw_set_color(c_white); // mantém a cor do texto branca
	}

	draw_text_transformed(gui_largura / 2, gui_altura / 2 + altura_str * i, menu_inicial[i], escala[i], escala[i], 0); // desenha o texto do item atual do menu com as transformações de escala aplicadas
}

// Resetando fonte e alinhamento
draw_set_halign(-1); // remove o alinhamento horizontal
draw_set_valign(-1); // remove o alinhamento vertical
draw_set_font(-1); // remove a fonte
