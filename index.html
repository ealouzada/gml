	Abaixo existe dois códigos GML para o menu do game test que foi criado recentemente (e publicado aqui).
	Existe um problema de interação entre 'mouse' e 'teclado'.

	Só é preciso deixar as funções já existentes de mouse e teclado 'somadas e funcionais'.
		

	/// @description Menu Criar

	window_set_size(1280, 720)

	options = ["novo jogo","carregar jogo","opções","sair"];
	index = 0;
	op_max = array_length(options);
		

	/// @description Menu Draw

	var _mx = device_mouse_x_to_gui(0); // MOUSE
	var _my = device_mouse_y_to_gui(0); //MOUSE

	draw_set_font(ftMenu);
	draw_set_halign(fa_center);
	draw_set_valign(fa_middle);

	var tam_menu = array_length(menu_inicial);
	for(var i = 0; i < tam_menu; i++)
	{
		var gui_largura = display_get_gui_width();
		var gui_altura = display_get_gui_height();

		var altura_str = string_height("I");
		var largura_str = string_width(menu_inicial[i]);

		var x1 = gui_largura / 2 - largura_str / 2;
		var y1 = gui_altura / 2 - altura_str / 2 + altura_str * i //top left MOUSE

		var x2 = gui_largura / 2 + largura_str / 2;
		var y2 = gui_altura / 2 + altura_str / 2 + altura_str * i // top right MOUSE

		if(point_in_rectangle(_mx, _my, x1, y1, x2, y2)) // MOUSE
		{
			esc[i] = lerp(esc[i], 1.4, 0.15);

			if(mouse_check_button_pressed(mb_left))
			{
				switch menu_inicial[i]
				{
					case menu_inicial[0]: //Jogar
						/*show_message("Vamos Jogar");*/
						room_goto(rLevel1)
						break;
					case menu_inicial[1]: //Carregar
						show_message("Carregar um jogo salvo");
						break;
					case menu_inicial[2]: //Opções
						show_message("Tenho que configurar algumas coisas");
						break;
					case menu_inicial[3]: //Sair
						game_end();
						break;
				}
			}

		}
		else
		{
			esc[i] = lerp(esc[i], 1, 0.15);
		}

		if(index == i)
		{
			draw_set_color(c_orange);
		}
		else
		{
			draw_set_color(c_white);
		}

		draw_text_transformed(gui_largura / 2, gui_altura / 2 + altura_str * i, menu_inicial[i], esc[i], esc[i], 0);
	}

	draw_set_halign(-1);
	draw_set_valign(-1);
	draw_set_font(-1);
